# Having

When we use group by we can't use where condition because group is below query instead of where we use having query after that group by.. having query filter Group by query

SELECT client_id,
SUM(invoice_total) as total_sales
FROM invoices
-- WHERE total_sales > 500 Where condition not work with group by
-- we have to use having instead of where below to Group by
GROUP BY client_id
HAVING total_sales > 500

-- With and

SELECT client_id,
SUM(invoice_total) as total_sales
COUNT(*) AS number_of_invoices
FROM invoices
-- WHERE total_sales > 500 Where condition not work with group by
-- we have to use having instead of where below to Group by
GROUP BY client_id
HAVING total_sales > 500 AND number_of_invoices > 5;

# SUBQUERIES IN THE FROM


-- better use views it's very big query make short with view below example of view

SELECT * FROM (
    SELECT
    client_id,
    name,
    (SELECT SUM(invoice_total) FROM invoices WHERE client_id = c.client_id) AS total_sales,
    (SELECT AVG(invoice_total) FROM invoices) AS average,

# IFNULL and COALESCE

If value null show this text

SELECT
    order_id,
    -- if shipper_id null show Not assigned
    IFNULL(shipper_id, 'Not assigned') AS shippers
FROM orders


-- COALESCE check two condition if both false print not assigned
SELECT
    order_id,
    -- if shipper_id null show Not assigned
    -- if shipper_id null show comments if comments is null show Not assigned
    COALESCE(shipper_id, comments,  'Not assigned') AS shippers
FROM orders

# IF FUNCTION

IF(expression, first, second) true first false second we don't need use union operator we can handle by if function

SELECT order_id, order_date
    IF(YEAR(order_date) = YEAR(NOW()), 'Active','Archived')
FROM orders

-- another example
SELECT product_id, name,
        COUNT(*) AS orders,
        IF(COUNT(*) > 1, 'Many times', 'Once') AS frequerince
FROM products
JOIN order_items USING (product_id)
GROUP BY product_id, name

# CASE OPERATOR

same like switch operator

SELECT order_id,

        ELSE 'Future'
    END AS category -- alias name
FROM orders
