function generateTimes(startTime, endTime, interval) {
  let times = [];
  let current = new Date();
  //   console.log(startTime.split(":")[0]);
  current.setHours(startTime.split(":")[0], startTime.split(":")[1], 0, 0); // Set start time
  let end = new Date();
  console.log(current);
  end.setHours(endTime.split(":")[0], endTime.split(":")[1], 0, 0); // Set end time
  while (current <= end) {
    // Push current time formatted as HH:MM
    times.push(current.toTimeString().substring(0, 5)); // Add interval
    current.setMinutes(current.getMinutes() + interval);
  }
  return times;
}

let times = generateTimes("08:00", "12:00", 30);
console.log(times);
