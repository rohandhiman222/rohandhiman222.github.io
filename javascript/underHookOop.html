function DogPro(name, breed) {
  this.name = name;
  this.breed = breed;
}

DogPro.prototype.bark = function () {
  return `${this.name} says woof`;
};

DogPro.prototype.sleep = function () {
  return `${this.name} is sleeping!`;
};

const dogPro = new DogPro("vodaphone", "indian");
const dogPro2 = new DogPro("vodaphone3", "indian");

console.log(dogPro, dogPro2);

console.log(dogPro.__proto__); // { bark: [Function (anonymous)], sleep: [Function (anonymous)] }

console.log(dogPro.sleep === dogPro2.sleep); // output true

            # Prototype chain

            // prototype is a object just
const grandParent = {
  greet() {
    return "Hii there";
  },
};

const parent = {
  color: "purple",
  sing() {
    return "La la la";
  },
  __proto__: grandParent,
};

const child = {
  num: 2,
  __proto__: parent,
};

console.log(child); // { num: 2 }

console.log(child.__proto__); // { color: 'purple', sing: [Function: sing] }

console.log(child.color); // purple

// first check child after parent after grandparent inside this prototype chain
console.log(child.greet()); // Hii there

console.log(child.__proto__.__proto__); // { greet: [Function: greet] }

            # Object

            const person = {
  sing() {
    return "hehhehe";
  },
  isHuman: true,
};

console.log(person); // { sing: [Function: sing], isHuman: true }
console.log(Object.create(person)); // {}
const tom = Object.create(person);
console.log(tom); //{}

console.log(Object.getPrototypeOf(tom)); // { sing: [Function: sing], isHuman: true }
