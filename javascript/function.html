# Pure functions

Referential transparency

  • The function always gives the same return value for the same arguments.

  • The function can't depend on any mutable state

Side-effect free:

  • The function can't cause any side effects.
  • Side effects any include I/O (eg, writing to the console or a log file), modifying a mutable object, reassigning a variable, etc.

Pure functions are fundamental to functional programming.

They allow us to write predictable and simple code eventually one day.

And the concept of a pure function is that pure functions are deterministic for a given input.

A pure function needs to always return the same value for that input, so same input gets the same output.

// impure function
let value = 0;
function squareAndUpdateValue(num) {
  const value = num * num;
  return value;
}

squareAndUpdateValue(3);

// second
const colors = ["red", "blue", "orange"];

function addToArray(arr, value) {
  return arr.push(value);
}

addToArray(colors, "black");

// pure
function square(num) {
  return num * num;
}

// second
const colors = ["red", "blue", "orange"];
function pureAddToArray(arr, value) {
  return [...arr, value];
}

console.log(pureAddToArray(colors, "black"));

# return

Higher order function

A function that recieves another function as an argument, returns a function, or does both.

// this is higher order function

function doTwice(func) {
  func();
  func();
}

doTwice(function () {
  console.log("Hello world Here");
});

// return a function
// it's work like factory function
function multiplyBy(factor) {
  return function (number) {
    return number * factor;
  };
}

const triple = multiplyBy(3);
const double = multiplyBy(3);
console.log(triple(10));
console.log(double(10));

# Partial application

The process of executing a function with some or all of its arguments. the partially applied function then gets returned for later use.

you've seen this idea before with the bind function!

function greet(greeting, name) {
  console.log(`${greeting}, ${name}!!!`);
}

const greetFun = greet.bind(null, "how are you ");
greetFun("rohan"); // how are you , rohan!!!

const greetFunHate = greet.bind(null, "I hate you ");
greetFunHate("rohan"); // I hate you , rohan!!!

// another way
function multiply(a, b) {
  return a * b;
}

const double = multiply.bind(null, 2);
console.log("double", double(2));
const triple = multiply.bind(null, 3);
console.log("triple", triple(3));

//////////// another exmple with function
function partial(func, ...fixedArgs) {
  return function (...remainingArgs) {
    return func(...fixedArgs, ...remainingArgs);
  };
}

const double2 = partial(multiply, 2);
const triple2 = partial(multiply, 3);

console.log("double 2", double2(5)); // 10
console.log("triple 2", triple2(5)); // 15

# Composition

function composition is a mechanism of combining multiple functions to build a more complicated one.

the result of each function is passed to the next one.

In Math, this can be expressed as f(g(x))

const add = (a, b) => a + b;
const multiple = (a, b) => a * b;
console.log(add(10, multiple(10, 20))); // 210

//////// another

const add2 = (a, b) => a + b;
const square = (a) => a * a;

console.log(square(add2(1, 3))); // 16
// 1+3 = 4 * 4 = 16

// 2*2 = 4, 3*3 = 9 === 13
console.log(add2(square(2), square(3))); //13

// third

const addAndSquare = (a, b) => square(add(a, b));

console.log(addAndSquare(10, 2)); // firsst 10+2 = 12*12 == 144
// 11*10 == 100, 4

When composing functions togethers, it's often best to not use methods, but instead create independent functions with a minimal number of arguments

function lowerCaseString(str) {
  return str.toLowerCase();
}

function splitWords(str) {
  return str.split(" ");
}

function joinWithDash(arr) {
  return arr.join("-");
}

console.log(joinWithDash(splitWords(lowerCaseString("My Favorite Game")))); // my-favorite-game

function compose(fn1, fn2) {
  return function (value) {
    return fn2(fn1(value));
  };
}

function repeatTwice(str) {
  return str.repeat(2);
}

function upperCase(str) {
  return str.toUpperCase();
}

console.log(compose(repeatTwice, upperCase)("Awesome")); //AWESOMEAWESOME

one more example

function compose(...functions) {
  return function (data) {
    return functions.reduceRight((val, func) => func(val), data);
  };
}

function lowerCaseString(str) {
  return str.toLowerCase();
}

function splitWords(str) {
  return str.split(" ");
}

function joinWithDash(arr) {
  return arr.join("-");
}

const sluggify = compose(joinWithDash, splitWords, lowerCaseString);
console.log(sluggify("Hello Man How are you?")); // hello-man-how-are-you?

# currying

A curried function can be called with any number of arguments- if you call it with fewer args then it takes, it returns a 'smaller' partial, which you can then call with remaining arguments.

f(a,b,c) f(a)(b)(c)

function add(a, b, c) {
  return a + b + c;
}

function addCurry(a) {
  return function (b) {
    return function (c) {
      return a + b + c;
    };
  };
}

console.log(addCurry(2)(4)(6)); // 12

///////////////////

function add(x, y, z) {
  return x + y + z;
}

function curry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    } else {
      return function (...args2) {
        return curried.apply(this, args.concat(args2));
      };
    }
  };
}

const curriedAdd = curry(add); // function add inside

console.log(curriedAdd(3, 4, 5)); // 12
console.log(curriedAdd(3)(4)(5)); // 12
