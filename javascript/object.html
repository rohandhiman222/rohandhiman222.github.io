we use super keyword for this

super.describe() like this.

class Triangle {
  constructor(a, b) {
    if (!Number.isFinite(a) || a <= 0) {
      throw new Error("Invalid a: " + a);
    }
    if (!Number.isFinite(b) || b <= 0) {
      throw new Error("Invalid b: " + b);
    }

    this.a = a;
    this.b = b;
  }

  getArea() {
    return (this.a + this.b) / 2;
  }

  getHypotenuse() {
    return Math.sqrt(this.a ** 2 + this.b ** 2);
  }
  describe() {
    return "Runs and hides tringle";
  }
}

class ShyTriangle extends Triangle {
  describe() {
    // use super key word for it
    const parentClassResult = super.describe();
    return "Runs and hides shytriangle " + parentClassResult;
  }
}

// automatic super key word ans pass value inside parent class so we don't need to add super key word in triangle class
let shy = new ShyTriangle(10, 20);
console.log(shy.describe()); // Runs and hides shytriangle Runs and hides tringle
console.log(shy.getArea());

# Super Key word

class Triangle {
  constructor(a, b) {
    this.a = a;
    this.b = b;
  }

  getArea() {
    return (this.a * this.b) / 2;
  }

  getHypotenuse() {
    return Math.sqrt(this.a ** 2 + this.b ** 2);
  }

  describe() {
    return `Area is ${this.getArea()}.`;
  }
}

class ColorTriangle extends Triangle {
  // we we want to add owr properties color we have to use super key word and pass value
  // if don't use constructor automatic pass super
  constructor(a, b, color) {
    super(a, b);
    this.color = color;
  }
}

class ColorMoodTriangle extends ColorTriangle {
  // we have to use same super key word and pass same value which are exetends
  constructor(a, b, color, mood) {
    super(a, b, color);
    this.mood = mood;
  }
}

# Static Properties

class Cat {
  constructor(name, breed) {
    this.name = name;
    this.breed = breed;
    // this.species = "felis catus"; use it in static best practice
  }

  // you don't need change every intance you can change it every instance with change
  static species = "felis catus";

  getInsideFun() {
    return "static Property " + Cat.species;
  }
}

// const cat = new Cat(); // Cat { name: undefined, breed: undefined }
const cat = new Cat("blue", "Scottish fold"); //Cat { name: 'blue', breed: 'Scottish fold' }

// static values is not showing inside cat instance inside
console.log(cat); // Cat { name: 'blue', breed: 'Scottish fold' }

// you can get like this
console.log(Cat.species); // felis catus
console.log(cat.getInsideFun()); // static Property felis catus

// all instance change automatic don't need one by one change
Cat.species = "i changed for all this class";

console.log(cat.getInsideFun()); // static Property i changed for all this class and instance

class Cat {
  constructor(name, breed) {
    this.name = name;
    this.breed = breed;
  }

  static registerStray() {
    const names = ["Muffin", "Biscuit", "Sleepy", "Dodo", "Princess"];

    const name = choice(names);
    return new Cat(name, "unknown");
  }
}

function choice(arr) {
  const randIdx = Math.floor(Math.random() * arr.length);
  return arr[randIdx];
}

console.log(Cat.registerStray()); // Cat { name: 'Dodo', breed: 'unknown' }

# Getters new features

JavaScript provides special methods, termed "getters" and "setters", which allow you define the ways to retrieve or change the values of object properties respectively.

Allow you to retrieve the value of an object's property.

class Circle {
  constructor(radius) {
    this._radius = radius;
  }

  // getter for the diameter
  // work as properties
  get diameter() {
    return this._radius * 2;
  }

  get radius() {
    return this._radius;
  }
}
const circle = new Circle(5);
console.log(circle.diameter); // 10 // working without () working like properties
console.log(circle.radius); // 5 // working without () working like properties

# Setters

Allow you to set the value of an object's property

class Circle {
  constructor(radius) {
    this._radius = radius;
  }

  // getter for the diameter
  // work as properties
  get diameter() {
    return this._radius * 2;
  }

  // work same like properties
  set radius(value) {
    if (value < 0) {
      throw new Error("Radius can't be negative");
    } else {
      this._radius = value;
    }
  }
}
const circle = new Circle(5);
console.log(circle.diameter); // 10 // working without () working like properties
circle.radius = 10; // same work set properties like properties
console.log(circle.diameter); // 10
