function createInputDom({ label, type = "text" }) {
  const labelEl = document.createElement("label");
  const inputEl = document.createElement("input");

  inputEl.type = type;
  labelEl.innerText = label;
  labelEl.append(inputEl);

  return labelEl;
}

  console.log(event.target);

function eventHandle(event) {
  // stopPropagation help you stop capturing
  //   event.stopPropagation();

  // stopImmdiatePropagation stop all event only single event work
  //   event.stopImmediatePropagation();
  console.log(event.target);
}

// event bubbling if click any class same response return
// this called event capturing so working like this, head,body,body,one,two, three got event
one.addEventListener("click", eventHandle);
two.addEventListener("click", eventHandle);
three.addEventListener("click", eventHandle);
// true and {capture:true} both are same
// three.addEventListener("click", eventHandle, true);
three.addEventListener("click", eventHandle, { capture: true });

// this event also work stopPropagation not stop all events

// state
let todos = [];

// selects
const form = document.forms.todos;
const input = form.elements.todo;
const root = document.querySelector(".todos");
const list = root.querySelector(".todos-list");
const count = root.querySelector(".todos-count");
// functions

function renderTodo(todos) {
  let todoString = "";
  todos.forEach((todo, index) => {
    todoString += `<li data-id="${index}"${
      todo.complete ? 'class="todos-complete"' : ""
    }>
        <input  type="checkbox"${todo.complete ? " checked" : ""} />
        <span>${todo.label}</span>
        <button type="button">X</button>
    </li><hr>`;
  });
  list.innerHTML = todoString;
  count.innerText = todos.filter((todo) => !todo.complete).length;
}

function addTodo(event) {
  event.preventDefault();
  const label = input.value.trim();
  const complete = false;
  // ... array automatic add next item in next array
  todos = [...todos, { label, complete }];
  console.log(todos);
  renderTodo(todos);
  input.value = "";
}

function updateTodo(event) {
  console.log(event.target);
  console.log(event.target.parentNode);
  const id = parseInt(event.target.parentNode.getAttribute("data-id"));
  console.log(id);
  const complete = event.target.checked;
  todos = todos.map((todo, index) => {
    if (index === id) {
      return {
        ...todo,
        complete,
      };
    }
    return todo;
  });
  console.log(todos);
  renderTodo(todos);
}

// init
function init() {
  // add todo
  form.addEventListener("submit", addTodo);
  // update todo
  list.addEventListener("change", updateTodo);
}
init();
