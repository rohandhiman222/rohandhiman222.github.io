# FlexBasis, FlexGrow and flexShrink

FlexBasis same width flexGrow same flex: 1 flexShrink don't cross width limit auto matic shrink

export default class App extends Component {
  render() {
    return (
      <View
        style={{
          backgroundColor: "#fff",
          flex: 1,
          flexDirection: "row",
          justifyContent: "center",
          alignItems: "center",
        }}
      >
        <View
          // flex basis same width : 100
          style={{
            backgroundColor: "dodgerblue",
            flexBasis: 100,
            height: 100,
          }}
          // flex grow mean same result flex: 1

          style={{
            backgroundColor: "dodgerblue",
            flexGrow: 1,
            height: 100,
          }}
          // flex shrink mean same auto adjust width 400 don't go outside

          style={{
            backgroundColor: "dodgerblue",
            width: 400,
            flexShrink: 1, // width: 400 automatic adjust and shrink
            height: 100,
          }}
        />

        <View
          style={{
            backgroundColor: "gold",
            width: 100,
            height: 100,
          }}
        />
        <View
          style={{
            backgroundColor: "tomato",
            width: 100,
            height: 100,
          }}
        />
      </View>
    );
  }
}
