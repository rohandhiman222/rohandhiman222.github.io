# Button

import { SafeAreaView, StyleSheet, Text, View } from "react-native";
import * as React from "react";
import { RectButton } from "react-native-gesture-handler";

import StyleGuide from "./StyleGuide";

interface ButtonProps {
  label: string;
  primary?: boolean;
  onPress: () => void;
}

const ButtonC = ({ label, primary, onPress }: ButtonProps) => {
  const color = primary ? "white" : undefined;
  const backgroundColor = primary ? StyleGuide.palette.primary : undefined;
  const padding = 33;
  return (
    <RectButton {...{ onPress }}>
      <SafeAreaView style={{ backgroundColor }} accessible>
        <View style={styles.container}>
          <Text style={[styles.label, { color }]}>{label}</Text>
        </View>
      </SafeAreaView>
    </RectButton>
  );
};

export default ButtonC;

const styles = StyleSheet.create({
  container: {
    padding: StyleGuide.spacing * 2,
  },
  label: {
    textAlign: "center",
  },
});

# Map array and add, delete

import { Text, StyleSheet, View } from "react-native";
import React, { Component } from "react";
import Navbar from "./src/components/Navbar";

export default class App extends Component {
  state = {
    random: [34, 31],
  };

  changeName = () => {
    const random = Math.floor(Math.random() * 100);
    this.setState((prevState) => {
      return {
        random: [...prevState.random, random],
      };
    });
  };

  delete = (position) => {
    // alert(position)
    const newArray = this.state.random.filter((item, index) => {
      return position != index;
    });

    this.setState({
      random: newArray,
    });
  };

  render() {
    return (
      <View>
        <Navbar
          random={this.state.random}
          nameHandle={this.changeName}
          delete={this.delete}
        />
      </View>
    );
  }
}

const styles = StyleSheet.create({});

import { Button, StyleSheet, Text, Touchable, View } from "react-native";

  return (
    <View>
      {props.random.map((item, id) => (

          </Text>
        </TouchableHighlight>
      ))}

      <Button title="Change Name" onPress={() => props.nameHandle()} />
    </View>
  );
};

export default Navbar;

/* Input Component */

import {
  Text,
  StyleSheet,
  View,
  TextInput,
  Button,
  ScrollView,
} from "react-native";
import React, { Component } from "react";

export default class Input extends Component {
  state = {
    name: "",
    users: ["rohan", "sonali"],
  };

  onInputChange = (event) => {
    this.setState({
      name: event,
    });
  };

  addHandle = () => {
    this.setState((prevState) => {
      return {
        name: "",
        users: [...prevState.users, prevState.name],
      };
    });
  };

  render() {
    return (
      <ScrollView>
        <TextInput
          style={{
            backgroundColor: "#808080",
            padding: 12,
            width: "100%",
            marginBottom: 10,
          }}
          value={this.state.name}
          onChangeText={this.onInputChange}
        />
        <Button
          title="Add"
          onPress={() => {
            this.addHandle();
          }}
        />
        {this.state.users.map((item, index) => (
          <Text
            key={index}
            style={{ backgroundColor: "#fafa55", padding: 10, margin: 10 }}
          >
            {item}
          </Text>
        ))}
      </ScrollView>
    );
  }
}

const styles = StyleSheet.create({});
