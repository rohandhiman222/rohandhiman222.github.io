                                                                  # Understanding Closure.

                                                                  • A closure is the local variables for a function- kept alive after that function has returned.

                                                                  • Closure is when a function is able to remember and access its lexical scope even when that function is executing outside its lexical scope.

                                                                  • A closure is a function having access to the parent scope, even after that parent function has closed.

                                                                  # Es6 Class

                                                                  Class Greeting {
    constructor(mGreet, nGreet){
        this.morningGreet = mGreet;
        this.eveningGreet = nGreet;
        this.dayTime = "morning";
    }

    greet() {
        if (this.dayTime === 'morning') {
            return this.morningGreet;
        }else if(this.dayTime === 'evening'){
            return this.eveningGreet;
        }else{
            return this.greeting;
        }
    }
}

class CommonGreeting extends Greeting {
    constructor(mGreet, nGreet){
        super(mGreet, nGreet);
        this.greeting = "Howdy";
        this.dayTime = "day"
    }
}

var aGreet = new CommonGreeting("Morning","Evening")

console.log(aGreet.greet() + " neighbor.");

aGreet.dayTime = "morning";
console.log(aGreet.greet() + " neighbor.");

                                                                  # Pure function

                                                                  • The function depends on the input provided and not on the external data that changes.

                                                                  • the function doesn't cause side effects. it doesn't cause change beyond its scope.

                                                                  • Given the same input, the function always return the same output.

                                                                  let inc = function(num){
    return num + 1; // this is pure function 1 never change 
}


const val = 5;

let multBy5 = function(num){
    return num * val; // this is also pure function val never change because it's const value
}

                                                                  # Factorial

                                                                  const factorial = function(num){
    if (num === 1) {
        return 1;
    }
    return num * factorial(num -1);
}

let final = factorial(5);
console.log(final);
