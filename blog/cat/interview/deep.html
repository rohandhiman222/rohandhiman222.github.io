function myPolyfilllFilter(arr, cb) {
  let filterArr = [];
  for (let i = 0; i < arr.length; i++) {
    if (cb(arr[i])) {
      filterArr.push(arr[i]);
    }
  }
  return filterArr;
}

Function.prototype.myBind = function (...args) {
  let obj = this;
  let params = args.slice(1);
  return function (...args2) {
    obj.apply(args[0], [...params, ...args2]);
  };
};

let showDetails = function (city, state) {
  console.log(
    this.name + " " + this.age + " " + " city " + city + " State " + state
  );
};
let showDetailsMyBind = showDetails.myBind(person1, "chd");

console.log(showDetailsMyBind("hr"));
