              # Object literal

              // object literal create object
const circle = {
  radius: 1,
  location: {
    x: 1,
    y: 1,
  },
  draw: function () {
    console.log("draw");
  },
};

circle.draw();

              # Factories

              // we have many circle so we can't use object literal  solution factories

function createCircle(radius) {
  return {
    radius: radius,
    draw: function () {
      console.log("draw");
    },
  };
}

const circle = createCircle(1);
circle.draw();

              # constructor

              // first latter should be capital
function Circle(radius) {
  this.radius = radius;
  this.draw = function () {
    console.log("draw");
  };
}

const another = new Circle(1);
console.log(another.constructor);

let number = 10;
function increase(number) {
  number++;
}
increase(number);
console.log(number); // 10

//

let obj = { value: 10 };

function increase(obj) {
  obj.value++;
}

// object pass by reference
increase(obj);
console.log(obj); // 11
