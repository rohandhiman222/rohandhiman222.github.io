                                                                const requestManager = (url, options = {}, attempts = 3) => {
  return new Promise((res, rej) => {
    fetch(url, options)
      .then(res)
      .catch((error) => {
        const isLastAttempt = attempts === 1;
        if (isLastAttempt) {
          return rej(error);
        }
        setTimeout(() => {
          requestManager(url, options, attempts - 1)
            .then(res)
            .catch(rej);
        }, 3000);
      });
  });
};

const typeOf = (input) => {
    const newObject = Object.prototype.toString.call(input).toLowerCase();
    const typeOfRegex = /\[object (.*)]/g;
    const type = typeOfRegex.exec(newObject)[1];
    return type;
}

const shallowCompare = (source, target) => {
    if (typeOf(source) !== typeOf(target)) {
        return false;
    }

    if (typeOf(source) === "array") {
        if (source.length !== target.length) {
            return false;
        }
        return source.every((el,index) => el === target[index]);
    }

    if (typeOf(source) === "object") {
        if (Object.keys(source).length !== Object.keys(target).length) {
            return false;
        }
        return Object.keys(source).every((key) => source[key] === target[key]);
    }

    if (typeOf(source) === "date") {
        return source.getTime() === target.getTime();
    }

    return source === target;


}


// it will not compare nested element
shallowCompare({a:1},{a:1}) // true

shallowCompare([1,2],[1,2]) // true

const deepCompare = (source, target) => {
    if (typeOf(source) !== typeOf(target)) {
        return false;
    }

    if (typeOf(source) === "array") {
        if (source.length !== target.length) {
            return false;
        }
        return source.every((el,index) => deepCompare(el,target[index]));
    }

    if (typeOf(source) === "object") {
        if (Object.keys(source).length !== Object.keys(target).length) {
            return false;
        }
        return Object.keys(source).every((key) => deepCompare(source[key], target[key]));
    }

    if (typeOf(source) === "date") {
        return source.getTime() === target.getTime();
    }

    return source === target;


}

                                                                # Memoization function design

                                                                const memoizeAdd = () => {
  let cache = {};
  return (value) => {
    if (value in cache) {
      console.log("fetching from cache");
      return cache[value];
    } else {
      console.log("calculating value");
      const result = value + 10;
      cache[value] = result;
      return result;
    }
  };
};

const newAdd = memoizeAdd();
console.log(newAdd(9)); // calculated 19
console.log(newAdd(9)); // cache

                                                                const fibonacci = (n) => {
  // 0,1,1,2,3,5,8,13

  if (n < 2) {
    return 1;
  } else {
    return fibonacci(n - 2) + fibonacci(n - 1);
  }
};

                                                                const isAnagram = (str1, str2) => {
  if (str1.length !== str2.length) {
    return false;
  }

  const lowerStr1 = str1.toLowerCase();
  const lowerStr2 = str2.toLowerCase();

  if (lowerStr1 === lowerStr2) {
    return false;
  }

  const sortedStr1 = lowerStr1.split("").sort().join();
  const sortedStr2 = lowerStr2.split("").sort().join();

  return sortedStr1 === sortedStr2;
};

isAnagram("listen", "silent");

                                                                # 12 to 24 time change

                                                                const convertTo24HoursFormat = (timeText) => {
  const timeTextLower = timeText.toLowerCase();
  let [hours, mins] = timeTextLower.split(":");
  if (timeTextLower.endsWith("am")) {
    hours = hours === "12" ? "0" : hours;
  } else if (timeTextLower.endsWith("pm")) {
    hours = hours === "12" ? hours : String(+hours + 12);
  }

  return hours.padStart(2, 0) + ":" + mins.slice(0, -2).padStart(2, 0);
};

console.log(convertTo24HoursFormat("5:00Am"));
console.log(convertTo24HoursFormat("12:33Pm"));
console.log(convertTo24HoursFormat("11:10Pm"));
