# by splice method update object

let users = [
  { id: 1, username: "user1", password: "pass1", age: 28 },
  { id: 2, username: "user2", password: "pass2", age: 35 },
  { id: 3, username: "user3", password: "pass3", age: 22 },
  { id: 4, username: "user4", password: "pass4", age: 30 },
  { id: 5, username: "user5", password: "pass5", age: 24 },
  { id: 6, username: "user6", password: "pass6", age: 29 },
  { id: 7, username: "user7", password: "pass7", age: 31 },
  { id: 8, username: "user8", password: "pass8", age: 27 },
  { id: 9, username: "user9", password: "pass9", age: 33 },
  { id: 10, username: "user10", password: "pass10", age: 26 },
];

let indexToUpdate = users.findIndex((obj) => obj.id === 5);

if (indexToUpdate !== -1) {
  // Create a new object with the updated properties
  let updatedObject = { ...users[indexToUpdate], username: "Updated Name" };

  // Use splice() to replace the old object with the updated one
  users.splice(indexToUpdate, 1, updatedObject);
}

console.log(users);

let arrayOfObjects = [
  { id: 1, name: "Alice" },
  { id: 2, name: "Bob" },
  { id: 3, name: "Charlie" },
];

// Method 1: Using filter()
let filteredArray = arrayOfObjects.filter((obj) => obj.id !== 2);
console.log("Method 1: Using filter()");
console.log(filteredArray);

// Method 2: Using splice() (Modifies the original array)
let indexToRemove = arrayOfObjects.findIndex((obj) => obj.id === 2);
if (indexToRemove !== -1) {
  arrayOfObjects.splice(indexToRemove, 1);
}
console.log("Method 2: Using splice()");
console.log(arrayOfObjects);

// Method 3: Using a loop to remove objects based on a condition
let newArray = [];
for (let i = 0; i < arrayOfObjects.length; i++) {
  if (arrayOfObjects[i].id !== 2) {
    newArray.push(arrayOfObjects[i]);
  }
}
console.log("Method 3: Using a loop");
console.log(newArray);
