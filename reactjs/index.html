# JSX template

We can't use direct jsx in js we have to import cdn babel babel help us js convert into jsx second we have to change script tag type text/babel

// importing cdn
 <script src="https://cdnjs.cloudflare.com/ajax/libs/babel-standalone/6.26.0/babel.min.js"></script>


// type we changed into babel
 <script type="text/babel">
      const myButton = <button type="button">Save</button>;

      ReactDOM.render(myButton, document.getElementById("app"));
</script>

Output

<div id="app">
  <button type="button">Save</button>
</div>

# setInterval

This is interval never stop it self we have to clear interval

let count = 0;
const interval = setInterval(() => {
  const paragraph = <p>{count}</p>;
  ReactDOM.render(paragraph, document.getElementById("app"));

  if (count === 1000) {
    clearInterval(interval);
  } else {
    count++;
  }
}, 30);

Output

<div id="app">
  <p>1000</p>
</div>

# Childrem props

const Paragraph = ({ text }) => <p>{text}</p>;

const Section = ({ headingText, children }) => (
  <section>
    <h2>{headingText} </h2>
    {children}
  </section>
);

const SectionBlock = () => (
  <Section headingText="My first blog post">
    <Paragraph text="My first post text" />
    <Paragraph text="My second paragraph" />
  </Section>
);

ReactDOM.render(<SectionBlock />, document.getElementById("app"));

Output

<div id="app">
  <section>
    <h2>My first blog post</h2>
    <p>My first post text</p>
    <p>My second paragraph</p>
  </section>
</div>
