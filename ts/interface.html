# Multiple Interface

enum Roles {
  admin = "admin",
  author = "author",
  editor = "editor",
}

interface Role {
  role: Roles.admin;
}

enum PermissionsList {
  read = "read",
  write = "write",
  execute = "execute",
}

interface UserPermissions {
  permissions: PermissionsList[];
}

interface User {
  name: string;
  email: string;
  phone: number;
}

interface UserWithAddress extends User {
  address: string;
}

const user: User = {
  name: "rohan",
  email: "rohan@gmail.com",
  phone: 9078698809,
};

interface AdminUser extends User, Role, UserPermissions {
  numberOfUsersReporting: number;
}

const userWithAddress: UserWithAddress = {
  name: "rohan",
  email: "rohan@gmail.com",
  phone: 23423432432,
  address: "This is address",
};

const adminUser: AdminUser = {
  name: "rohan",
  email: "rohan@gmail.com",
  phone: 9809898,
  role: Roles.admin,
  permissions: [
    PermissionsList.read,
    PermissionsList.write,
    PermissionsList.execute,
  ],
  numberOfUsersReporting: 20,
};

//   type: string; // many type convert generic

//   color: string[]; // may colors // convert generic
//   description: string;
// }

enum AutoMobileTypes {
  car = "car",
  bus = "bus",
  bike = "bike",
  van = "van",
  truck = "truck",
}

enum AutoMobileBrands {
  bmw = "bmw",
  honda = "honda",
  ferrari = "ferrari",
  toyota = "toyota",
}

enum AutoMobileColors {
  red = "red",
  blue = "blue",
  brown = "brown",
  black = "black",
  silver = "silver",
}

interface Automobile<Type, Brand, Color> {
  type: Type;
  brand: Brand;
  color: Color[];
  description: string;
}

class Car implements Automobile<AutoMobileTypes, string, string> {
  public type: AutoMobileTypes = AutoMobileTypes.car;

  constructor(
    public brand: AutoMobileBrands,
    public color: string[],
    public description: string
  ) {}
}

const ferrari = new Car(
  AutoMobileBrands.ferrari,
  ["red", "green"],
  "this is a farrier"
);

console.log(ferrari);

    public description: string
  ) {}
}

const truck = new Truck(
  AutoMobileBrands.bmw,
  [AutoMobileColors.black],
  "this is a truck"
);

interface Automobile<Type, Brand, Color> {

# Interfaces and ac cess Modifiers

Here is an important thing to remember about interfaces.

Interfaces are a contract only for the public members of a class.

Interfaces are a blueprint of the final object.

I repeat, interfaces are a blueprint of the final object that a class will create.

And since objects do not have any access modifiers, that applies to interfaces as well.
