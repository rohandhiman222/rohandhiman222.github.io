# Inheritance

class User {
  name: string;
  email: string;
  age: number;

  constructor(name: string, email: string, age: number) {
    this.name = name;
    this.email = email;
    this.age = age;
  }
}

class AdminUser extends User {
  isAdmin: boolean = true; // default value
}

const user: User = new User("rohan", "rohan@gmial.com", 23);

const admin: AdminUser = new AdminUser("admin", "admin@gmail.com", 32);

console.log(user, admin);

/* 
User { name: 'rohan', email: 'rohan@gmial.com', age: 23 }
AdminUser { name: 'admin', email: 'admin@gmail.com', age: 32, isAdmin: true }
*/

# Super() keyword

class User {
  name: string;
  email: string;
  age: number;

  constructor(name: string, email: string, age: number) {
    this.name = name;
    this.email = email;
    this.age = age;
  }
}

class AdminUser extends User {
  isAdmin: boolean = true; // default value
  usersReporting: number;
  constructor(
    name: string,
    email: string,
    age: number,
    usersReporting: number
  ) {
    super(name, email, age);
    this.usersReporting = usersReporting;
  }
}

const admin: AdminUser = new AdminUser("admin", "admin@gmail.com", 32, 100);

console.log(admin);

/* 
AdminUser {
  name: 'admin',
  email: 'admin@gmail.com',
  age: 32,
  isAdmin: true,
  usersReporting: 100
}*/

class Person {
  // by default public all properties
  name: string;
  age: number;

  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }
}

let person: Person = new Person("rohan", 31);
person.name = "name is hacked";

console.log(person); // Person { name: 'name is hacked', age: 31 }

/////// stop this hack behavier use by access moddifers

/* 
we have three types of access modifiers

Public =>
yes outside the class => public
yes child class => public
yes parent class => public

Protected => 
  No outside the class => protected
  yes child class => protected
  yes parent class => protected

Private => 
  No outside the class => private;
  no child class => private
  yes parent class => private  
*/

class Person {
  // by default public all properties
  // public name: string;
  // public age: number;

  // protected name: string;
  // protected age: number;

  private name: string;
  private age: number;

  constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }
}

# Getting more control on members of a class

validate on age 0 to 100 we can add condition on constructor but it's not good idea so we will use get and set method here

Note set method only accept one parameter

Note get method we can send any parameter

Note get and set method access like properties not method person.age not person.age()

class Person {
    // by default undefined
  private _age: number | undefined;

  constructor(protected name: string) {}

  public getName() {
    return this.name;
  }

  public set age(age: number) {
    if (age > 100 || age < 0) {
      throw new Error("Age should be 0 - 100");
    }

    this._age = age;
  }

  public get age() {
    if (this._age === undefined) {
      throw new Error("Age not set yet");
    }
    return this._age;
  }
}

const person: Person = new Person("rohan");
person.age = 10; // set age
console.log(person.age); // get age 10
