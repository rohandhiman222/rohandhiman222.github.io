# duplicate remove json file

let arr = require("./allsentences.json");

var clean = arr.filter(
  (arr, index, self) =>
    index ===
    self.findIndex(
      (t) => t.english.toLowerCase().trim() === arr.english.toLowerCase().trim()
    )
);

console.log("done");

const fs = require("fs");

fs.writeFile("data121.json", JSON.stringify(clean), function(err) {
  if (err) {
    return console.log(err);
  }
  console.log("The file was saved!");
});
