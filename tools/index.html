What is ssh?
* Secure Shell
* Communication Protocol (like http, https, ftp, etc)
* Do just about anything on the remote computer 
* Traffic is encrypted
* Used mostly in the termial/command line


Client/Server Communication
* SSH is the client
* SSHD is the server (Open SSH Daemon)
* The server must have sshd installed and running on you will be able to connect using ssh


Authentication Methods
> ssh USERNAME@192.168.1.29
* Password
* public / private key pair
* host based

# popup.js

$(function(){

    // first click extension get value from storage
    chrome.storage.sync.get(['total','limit'],function(budget){
        $('#total').text(budget.total);
        $('#limit').text(budget.limit);
    });

    $('#spendAmount').click(function(){
        // we are getting here two variable value of total  and limit
        chrome.storage.sync.get(['total', 'limit'],function(budget){
            // first time zero value
            var newTotal = 0;
            // checking here total have value or not 
            if (budget.total){
                // if we found value store in newTotal variable
                newTotal += parseInt(budget.total);
            }

            var amount = $('#amount').val();
            if (amount){
                newTotal += parseInt(amount);
            }


            // storing value inside store we need object

            chrome.storage.sync.set({'total': newTotal}, function(){               
                if (amount && newTotal >= budget.limit){
                //    creating notifcation 
                    var notifOptions = {
                        type: "basic",
                        iconUrl: "icon48.png",
                        title: "Limit reached!",
                        message: "Uh oh, look's like you've reached your alloted limit."
                };
                chrome.notifications.create('limitNotif', notifOptions);

            }
            });
            $('#total').text(newTotal);
            $('#amount').val('');

           

        });
    });
});

    });
});

# eventPage.js

background run this script

var menuItem = {
    "id": "spendMoney",
    "title": "Spend Money",
    "contexts": ["selection"]
};


// Create context Menus
chrome.contextMenus.create(menuItem);


// Check value is int or not
function isInt(value) {
  return !isNaN(value) && 
         parseInt(Number(value)) == value && 
         !isNaN(parseInt(value, 10));
}



// run function when click content Menu
chrome.contextMenus.onClicked.addListener(function(clickData){   
    
    // get clicked data
    if (clickData.menuItemId == "spendMoney" && clickData.selectionText){    
        if (isInt(clickData.selectionText)){          
            chrome.storage.sync.get(['total','limit'], function(budget){
                var newTotal = 0;
                if (budget.total){
                    newTotal += parseInt(budget.total);
                }

                newTotal += parseInt(clickData.selectionText);
                
                // set value in store
                chrome.storage.sync.set({'total': newTotal}, function(){               
                if (newTotal >= budget.limit){
                    var notifOptions = {
                        type: "basic",
                        iconUrl: "icon48.png",
                        title: "Limit reached!",
                        message: "Uh oh, look's like you've reached your alloted limit."
                    };

                    // sending notification
                    chrome.notifications.create('limitNotif', notifOptions);

                    }
                });
            });
        }
    }
});

// storage change event 
chrome.storage.onChanged.addListener(function(changes, storageName){
    chrome.browserAction.setBadgeText({"text": changes.total.newValue.toString()});
});
